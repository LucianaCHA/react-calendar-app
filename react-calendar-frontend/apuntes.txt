self.addEventListener('fetch', (event) =>{
  if(event.request.url !== 'http://localhost:4000/api/auth/renew' ) return

  // esto es network only
  const response = fetch(event.request)
  .then((response)=> {
    return response.clone()
  })

  event.respondWith(response)
